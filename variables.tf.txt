variable "region" {
  description = "The AWS region in which to create resources."
  type        = string
  default     = "us-east-1" # You can change the default region or set it explicitly when calling Terraform
}

variable "max_capacity" {
  description = "The maximum capacity for the scalable target."
  type        = number
  default     = 10
}

variable "min_capacity" {
  description = "The minimum capacity for the scalable target."
  type        = number
  default     = 1
}

variable "resource_id" {
  description = "The resource ID for the scalable target."
  type        = string
}

variable "scalable_dimension" {
  description = "The scalable dimension for the scalable target."
  type        = string
}

variable "service_namespace" {
  description = "The service namespace for the scalable target."
  type        = string
}

variable "adjustment_type" {
  description = "The adjustment type for the scaling policy."
  type        = string
}

variable "cooldown" {
  description = "The cooldown period for the scaling policy."
  type        = number
  default     = 300
}

variable "metric_aggregation_type" {
  description = "The metric aggregation type for the scaling policy."
  type        = string
  default     = "Average"
}

variable "step_adjustments" {
  description = "A list of step adjustments for the scaling policy."
  type        = list(object({
    metric_interval_lower_bound = number
    metric_interval_upper_bound = number
    scaling_adjustment          = number
  }))
}

variable "alarm_name" {
  description = "The name of the CloudWatch alarm."
  type        = string
}

variable "alarm_description" {
  description = "The description of the CloudWatch alarm."
  type        = string
}

variable "comparison_operator" {
  description = "The comparison operator for the CloudWatch alarm."
  type        = string
}

variable "evaluation_periods" {
  description = "The number of evaluation periods for the CloudWatch alarm."
  type        = number
  default     = 1
}

variable "metric_name" {
  description = "The name of the CloudWatch metric."
  type        = string
}

variable "namespace" {
  description = "The namespace of the CloudWatch metric."
  type        = string
}

variable "period" {
  description = "The period of the CloudWatch metric."
  type        = number
  default     = 60
}

variable "statistic" {
  description = "The statistic for the CloudWatch alarm."
  type        = string
}

variable "threshold" {
  description = "The threshold value for the CloudWatch alarm."
  type        = number
}

variable "dimensions" {
  description = "A list of dimension maps for the CloudWatch alarm."
  type        = list(object({
    name  = string
    value = string
  }))
}

variable "treat_missing_data" {
  description = "The treatment for missing data in the CloudWatch alarm."
  type        = string
  default     = "notBreaching"
}
